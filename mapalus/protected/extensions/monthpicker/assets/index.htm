Example 3, elements settings

	
$("#example_3").monthpicker({
	elements: [
		{tpl:"month",opt:{

			value: 1
		}},
		{tpl:"year",opt:{

			range: "-10~5",
			value: 2007
		}}

	],
	onChanged: callback
});

Example 4, quarter selector

	
$("#example_4").monthpicker({
	elements: [
		{tpl:"year",opt:{

			range: "-10~2",
			value: 2005
		}},

		{tpl:"quarter"}
	],
	onChanged: callback
});

Callback function

	
		function callback(data,$e){
	var str = "";

	for(key in data) {
		str += " " + key + ": " + data[key]+ "; ";

	}
 
	var selector = "#message_" + $e.attr('id');

	$(selector).prepend("{" + str + "};<br />").show();

}
